* Code Blocks (embedded highlighting)                                 :code:

#+BEGIN_SRC python
def greet(name: str) -> str:
    # Python comment — #+BEGIN_SRC line above gets comment color here
    return f"hello, {name}"
#+END_SRC

#+BEGIN_SRC bash
#!/usr/bin/env bash
# Shell comment
for f in *.org; do
    echo "processing $f"
done
#+END_SRC

#+BEGIN_SRC javascript
// JS comment
const greet = (name) => `hello, ${name}`;
console.log(greet("org-mode"));
#+END_SRC

#+BEGIN_SRC js
// alias: js → javascript lexer
fetch("/api").then(r => r.json()).then(console.log);
#+END_SRC

#+BEGIN_SRC lua
-- Lua comment
local function greet(name)
    return "hello, " .. name
end
#+END_SRC
